// utils/request.js
const BASE_URL = 'https://your-domain.com/api/v1'

const request = (url, method = 'GET', data = {}) => {
  return new Promise((resolve, reject) => {
    wx.request({
      url: BASE_URL + url,
      method,
      data,
      header: {
        'Content-Type': 'application/json',
        'Authorization': wx.getStorageSync('token') || ''
      },
      success: (res) => {
        if (res.data.code === 200) return resolve(res.data.data)
        if (res.data.code === 401) {
          wx.reLaunch({ url: '/pages/login/index' })
          return reject(res.data)
        }
        reject(res.data)
      },
      fail: reject
    })
  })
}

export const get = (url, data) => request(url, 'GET', data)
export const post = (url, data) => request(url, 'POST', data)
export const put = (url, data) => request(url, 'PUT', data)
export const del = (url) => request(url, 'DELETE')

## Phase 1：基础框架与登录

### 统一响应格式与全局异常

```java
@Data
public class Result<T> {
    private Integer code;
    private String message;
    private T data;

    public static <T> Result<T> ok(T data) {
        Result<T> r = new Result<>();
        r.code = 200; r.message = "success"; r.data = data;
        return r;
    }
    public static Result<?> fail(int code, String msg) {
        Result<?> r = new Result<>();
        r.code = code; r.message = msg;
        return r;
    }
}
```

全局异常处理器覆盖：参数校验（400）、业务异常（自定义 code）、未登录（401）、无权限（403）、其他（500 + 记日志）。

### 微信授权登录

**接口：** `POST /api/v1/auth/wx-login`

```
小程序 wx.login() → 获取 code
→ POST /api/v1/auth/wx-login { code }
→ 后端调用微信 code2session（appid + secret + code）
→ 获取 openid（session_key 用完即丢，不存库）
→ 查或建 member 记录
→ 生成 JWT（payload: memberId, openid, 过期时间）
→ 返回 { token, member }
```

JWT 有效期 7 天，前端每次请求校验有效期，剩余不足 1 天时静默刷新。

### 学号/账号登录

**接口：** `POST /api/v1/auth/account-login`

查 `member_account` 表，BCrypt 验证密码，成功后返回同样的 token 结构。

### 后台管理员登录

**接口：** `POST /api/v1/admin/auth/login`

后台使用独立 JWT，payload 中包含 `adminId` 和 `roleId`。所有 `/api/v1/admin/**` 接口由独立拦截器校验。

### RBAC 权限

角色权限枚举（存为 JSON 数组在 sys_role.permissions）：
```
news:read  news:write  news:publish
hall:read  hall:write
course:read  course:write
enroll:read  enroll:export
stats:view
admin:super  （包含所有权限）
```

接口注解：`@RequirePermission("news:publish")` → 拦截器校验当前角色的 permissions 是否包含该值。

## Phase 2：首页、公告、新闻、搜索

### 模块：新闻资讯

**建设目标：** 实现新闻的发布、分类浏览、详情展示、点赞收藏分享、智能推荐。

**涉及页面：** 首页推荐卡片、packageA/news/list、packageA/news/detail

**后台入口：** 新闻管理（列表、新建、编辑、分类管理、上下架、置顶、AI 辅助）

**数据表：** news、category、like_record、favorite、share_record、search_index

**接口清单：**
```
GET  /api/v1/news                     列表（分页、分类过滤、置顶优先）
GET  /api/v1/news/{id}                详情（+1 浏览量 Redis INCR）
POST /api/v1/news/{id}/like           点赞/取消（toggle，需登录）
POST /api/v1/news/{id}/favorite       收藏/取消（toggle，需登录）
POST /api/v1/news/{id}/share          记录分享行为
GET  /api/v1/news/{id}/related        相关推荐（同分类最近 5 条）
GET  /api/v1/admin/news               后台列表（分页+筛选）
POST /api/v1/admin/news               新建
PUT  /api/v1/admin/news/{id}          编辑
PUT  /api/v1/admin/news/{id}/publish  发布（触发 search_index 同步）
PUT  /api/v1/admin/news/{id}/unpublish 下架
POST /api/v1/admin/news/{id}/ai-polish AI 文案辅助
```

**权限规则：** 游客可读；点赞/收藏/分享需登录；后台需 news:write 权限。

**业务流程（发布）：** 编辑器保存草稿 → 提交发布 → 微信内容安全审核通过 → 更新状态为 published → 异步同步至 search_index。

**异常处理：** 内容安全未通过返回具体原因；已发布新闻不可直接删除，需先下架。

**统计/缓存规则：**
- 浏览量：Redis INCR，同用户 30 分钟内重复不计（键 TTL 30min），每 5 分钟批量落库
- 列表缓存：Redis 按分类缓存，TTL 5 分钟，发布/下架时失效
- 详情缓存：Redis 按 ID 缓存，TTL 10 分钟，编辑时失效

**验收标准：** 后台发布后小程序端立即可见；点赞数/收藏数实时更新；相关推荐展示同分类内容；分享可生成分享卡片。

public void generateSubtitle(Long courseId, String ossVideoUrl) {
    String taskId = asrClient.submitTask(ossVideoUrl);
    courseMapper.updateSubtitleStatus(courseId, "processing", taskId);

**报名核心逻辑（数据库事务为最终一致性保障）：**
```java
@Transactional
public void enroll(Long activityId, Long memberId, EnrollDTO dto) {
    // 1. 检查活动状态和报名时间
    Activity activity = activityMapper.selectById(activityId);
    if (!activity.canEnroll()) throw new BizException(409, "当前活动不可报名");

    // 2. 原子扣减名额（数据库事务兜底）
    int affected = activityMapper.incrEnrolledCount(activityId, activity.getQuota());
    if (affected == 0) throw new BizException(409, "名额已满");

    // 3. 插入报名记录（唯一索引防重复）
    try {
        Enroll enroll = buildEnroll(activityId, memberId, dto);
        enrollMapper.insert(enroll);
    } catch (DuplicateKeyException e) {
        activityMapper.decrEnrolledCount(activityId); // 回滚计数
        throw new BizException(409, "您已报名该活动");
    }

    // 4. 发送订阅消息（如有授权记录）
    subscribeService.send(memberId, "enroll_success", activity);
    // 5. 写站内消息（兜底）
    messageService.create(memberId, "报名成功", activity.getTitle());
}
```

```sql
-- mapper 层原子 SQL
UPDATE activity
SET enrolled_count = enrolled_count + 1
WHERE id = #{activityId}
  AND status = 'published'
  AND enrolled_count < quota;
```

**订阅消息发送：**
```java
public void send(Long memberId, String scene, Object data) {
    MemberSubscribeRecord record = subscribeMapper.findAvailable(memberId, scene);
    if (record == null || record.getAvailableCount() <= 0) return;
    try {
        wxService.sendSubscribeMessage(memberId, scene, data);
        subscribeMapper.decrAvailable(record.getId()); // 扣减次数
    } catch (Exception e) {
        log.warn("订阅消息发送失败，站内消息已兜底: {}", e.getMessage());
    }
}
```

**报名审核名额口径：** 若活动开启 `need_review`，用户提交报名后即占用名额（状态为 pending）；管理员拒绝或用户取消报名时释放名额。采用"提交即占位"策略，可避免多人同时待审导致超额。

**异常处理：** 报名未开始、报名已截止、名额已满、重复报名、活动已取消、个人信息不完整，均需返回具体提示。

**验收标准：** 正常报名成功；重复报名被拦截；名额满后拒绝；取消报名恢复名额；后台可查看/导出名单；有授权的用户收到微信提醒；无授权的用户可在消息中心查看通知。

```java
public void addPoints(Long memberId, String action) {
    PointRule rule = pointRuleMapper.findByAction(action);
    if (rule == null || !rule.isEnabled()) return;

    // 每日上限检查
    String dailyKey = "points:daily:" + memberId + ":" + action;
    Long todayCount = redis.incr(dailyKey);
    if (todayCount == 1) redis.expire(dailyKey, untilMidnight());
    if (todayCount > rule.getDailyLimit()) return;

    // 记录积分
    pointRecordMapper.insert(new PointRecord(memberId, action, rule.getPoints()));
    memberMapper.addPoints(memberId, rule.getPoints());

    // 检查是否解锁新徽章
    checkAndGrantBadge(memberId);
}
```

**验收标准：** 积分行为触发后余额正确增加；每日上限生效；达成条件时徽章自动颁发；徽章墙展示已获/未获状态。

public Result<AiMessageVO> chat(@PathVariable Long sessionId,
                                 @RequestBody AiChatDTO dto) {
    // 1. 输入安全审核
    if (!contentSafetyService.checkText(dto.getQuestion())) {
        return Result.fail(400, "问题包含不当内容，请修改后重试");
    }
    // 2. 每日次数限制
    checkDailyLimit(getCurrentMemberId());
    // 3. 知识库检索
    List<KnowledgeChunk> chunks = knowledgeService.retrieve(dto.getQuestion(), 5);
    // 4. 调用大模型（同步等待完整答案）
    String answer = aiService.chat(buildSystemPrompt(), buildUserPrompt(chunks, dto.getQuestion()));
    // 5. 输出安全审核
    if (!contentSafetyService.checkText(answer)) {
        answer = "该问题暂时无法回答，请换个方式提问。";
    }
    // 6. 存会话记录
    aiMessageMapper.insert(buildMessage(sessionId, dto.getQuestion(), answer, chunks));
    return Result.ok(buildVO(answer, chunks));
}

**余弦相似度检索：**
```java
public List<KnowledgeChunk> retrieve(String question, int topK) {
    float[] queryVec = embeddingService.embed(question);
    List<KnowledgeChunk> candidates = chunkMapper.findCandidates(
        extractKeywords(question), 200);
    return candidates.stream()
        .map(c -> { c.setScore(cosineSim(queryVec, parseVec(c.getEmbeddingJson()))); return c; })
        .sorted(Comparator.comparingDouble(KnowledgeChunk::getScore).reversed())
        .limit(topK).collect(toList());
}
```
